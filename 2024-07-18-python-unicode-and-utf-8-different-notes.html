with open('all_unicode_char.txt', 'wb') as f:
    err_nums = 0
    true_nums = 0
    for i in range(1, pow(2, 19)):
        try:
            s = chr(i).encode('utf8')
        except UnicodeEncodeError:
            err_nums += 1
            s = None
        if s:
            f.write(s)
            true_nums += 1
    print(
        'the true add error nums equals range(1, pow(2,19)):',
        (err_nums + true_nums) == len(range(1,pow(2,19)))
    )
